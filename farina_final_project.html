The function below is designed to verify whether the model passed as parameters exists under the given file name and, if it exists, the model is simply loaded, otherwise it is retrained from scratch. The batch size can also be specified alongside the number of epochs

In[11]:
def load_or_fit(model, model_file_name, batch_size = 64, epochs = 5):
    model_exists = os.path.exists(model_file_name)
    model_history = None
    if model_exists:
        model = load_model(model_file_name)
    else:
        model_history = model.fit(
            X_train,
            y_train,
            batch_size=batch_size,
            epochs=epochs,
            validation_data=(X_test, y_test)
        )
        model.save(model_file_name)
    return model, model_history

def parse_text(text):
    return [word_index[word] for word in text.lower().split()]

def evaluate_review(model, review):
    parsed_review = pad_sequences([parse_text(review)], maxlen=500, padding='post')
    # parsed_review = parse_text(review)
    prediction = model.predict(parsed_review, verbose=0)
    return prediction
